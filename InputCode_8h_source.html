116 [[nodiscard]] std::string to_string() const {
117 const char* kind_str = (kind == InputKind::Axis) ? "Axis" : "Button";
118
119 return std::visit([&](auto&& code_value) -> std::string {
120 using T = std::decay_t<decltype(code_value)>;
121 if constexpr (std::is_same_v<T, SDL_Scancode>) {
122 const char* name = SDL_GetScancodeName(code_value);
123 return std::string("Keyboard:") + (name ? name : "Unknown");
124 } else if constexpr (std::is_same_v<T, Uint8>) {
125 return std::string("Mouse") + kind_str + ":" + std::to_string(code_value);
126 } else if constexpr (std::is_same_v<T, SDL_GamepadButton>) {
127 const char* name = SDL_GetGamepadStringForButton(code_value);
128 return std::string("GamepadButton:") + (name ? name : "Unknown");
129 } else if constexpr (std::is_same_v<T, SDL_GamepadAxis>) {
130 const char* name = SDL_GetGamepadStringForAxis(code_value);
131 return std::string("GamepadAxis:") + (name ? name : "Unknown");
132 } else {
133 return "Unknown";
134 }
135 }, data);
136 }

138 static InputCode from_string(const std::string& str) {
139 const auto separator = str.find(':');
140 if (separator == std::string::npos) {
141 throw std::invalid_argument("Malformed InputCode string: " + str);
142 }
143
144 const std::string type_kind = str.substr(0, separator);
145 const std::string value = str.substr(separator + 1);
146
147 if (type_kind == "Keyboard") {
148 const SDL_Scancode code = SDL_GetScancodeFromName(value.c_str());
149 if (code == SDL_SCANCODE_UNKNOWN) {
150 throw std::invalid_argument("Unknown scancode: " + value);
151 }
152 return FromKeyboard(code);
153 }
154 if (type_kind == "MouseButton") {
155 return FromMouseButton(static_cast<Uint8>(std::stoi(value)));
156 }
157 if (type_kind == "MouseAxis") {
158 return FromMouseAxis(static_cast<Uint8>(std::stoi(value)));
159 }
160 if (type_kind == "GamepadButton") {
161 const SDL_GamepadButton button = SDL_GetGamepadButtonFromString(value.c_str());
162 if (button == SDL_GAMEPAD_BUTTON_INVALID) {
163 throw std::invalid_argument("Unknown gamepad button: " + value);
164 }
165 return FromGamepadButton(button);
166 }
167 if (type_kind == "GamepadAxis") {
168 const SDL_GamepadAxis axis = SDL_GetGamepadAxisFromString(value.c_str());
169 if (axis == SDL_GAMEPAD_AXIS_INVALID) {
170 throw std::invalid_argument("Unknown gamepad axis: " + value);
171 }
172 return FromGamepadAxis(axis);
173 }
174
175 throw std::invalid_argument("Unknown input type: " + type_kind);
176 }

195
196 // combine hashes using XOR and bit shift similar to boost's hash_combine
197 return code_hash ^ (kind_hash << 2) ^ (device_type_hash >> 2) ^ (data_hash << 4);
