42 template <typename T>
43 void register_component() {
44 auto type_index = std::type_index(typeid(T));
45 if (component_types_.find(type_index) != component_types_.end()) {
46 log::Logger::error("[ComponentManager] Component type already registered!");
47 return;
48 }
49 component_types_[type_index] = next_component_type_;
50 component_arrays_[type_index] = std::make_shared<DenseComponentArray<T>>();
51 ++next_component_type_;
52 }
53
54 template <typename T>
55 [[nodiscard]] bool is_registered() const {
56 auto type_index = std::type_index(typeid(T));
57 return component_types_.find(type_index) != component_types_.end();
58 }

60 template <typename T>
61 void add_component(Entity entity, const T& component) {
62 if (ComponentType component_type; !try_get_component_type<T>(component_type)) {
63 log::Logger::error("[ComponentManager] Tried to add unregistered component type.");
64 return;
65 }
66
67 if (has_component<T>(entity)) {
68 log::Logger::warn("[ComponentManager] Component already exists for this entity.");
69 return;
70 }
71
72 get_component_array<T>()->insert_data(entity, component);
73 }
74
75 template <typename T>
76 void remove_component(Entity entity) {
77 if (!has_component<T>(entity)) {
78 log::Logger::warn("[ComponentManager] Component doesn't exist for this entity.");
79 return; // safe to skip
80 }
81
82 auto component_array = get_component_array<T>();
83 component_array->remove_data(entity);
84 }
85
86 template <typename T>
87 T& get_component(Entity entity) {
88 if (!has_component<T>(entity)) {
89 sb::log::Logger::error("[ComponentManager] Component doesn't exist for this entity.");
90
91 // Throwing as this particular scenario could lead to very bad behavior
92 throw std::runtime_error("[ComponentManager] Tried to get nonexistent component.");
93 }
94
95 auto component_array = get_component_array<T>();
96 return component_array->get_data(entity);
97 }

99 template <typename T>
100 [[nodiscard]] bool has_component(Entity entity) const {
101 auto component_array = get_component_array<T>();
102 return component_array->has_data(entity);
103 }
104
105 void entity_destroyed(const Entity entity) override {
106 for (auto& [type_index, component_array] : component_arrays_) {
107 component_array->entity_destroyed(entity);
108 }

111 template <typename T>
112 [[nodiscard]] ComponentType get_component_type() const {
113 ComponentType component_type;
114
115 if (!try_get_component_type<T>(component_type)) {
116 log::Logger::error("[ComponentManager] Failed to get component type.");
117 return static_cast<ComponentType>(-1);
118 }
119
120 return component_type;
121 }

123 template <typename T>
124 [[nodiscard]] bool try_get_component_type(ComponentType& component_type) const {
125 auto type_index = std::type_index(typeid(T));
126 auto type = component_types_.find(type_index);
127
128 if (type == component_types_.end()) {
129 log::Logger::warn("[ComponentManager] Tried to get unregistered component type.");
130 return false;
131 }
132
133 component_type = type->second;
134 return true;
135 }

139 template <typename T>
140 DenseComponentArray<T>* get_component_array() {
141 return get_component_array_from_map<T>(component_arrays_);
142 }
143
144 template <typename T>
145 const DenseComponentArray<T>* get_component_array() const {
146 return get_component_array_from_map<T>(component_arrays_);
147 }
148
149 template <typename T, typename MapType>
150 static auto get_component_array_from_map(MapType& map) {
151 const auto type_index = std::type_index(typeid(T));
152 const auto it = map.find(type_index);
153
154 if (it == map.end()) {
155 log::Logger::error("[ComponentManager] Component type not registered before use.");
156 } else if (!it->second) {
157 log::Logger::error("[ComponentManager] Component storage pointer is null.");
158 }
159
160 // allows for code reuse for const and non-const contexts
161 using PointerType = std::conditional_t<std::is_const_v<MapType>, const DenseComponentArray<T>,
162 DenseComponentArray<T>>;
163 return static_cast<PointerType*>(it->second.get());
164 }
165
166 private:
167 std::unordered_map<std::type_index, ComponentType> component_types_;
168 std::unordered_map<std::type_index, std::shared_ptr<IComponentArray>> component_arrays_;
169 ComponentType next_component_type_ = 0;
170};
